Using Python (Recommended):

import pandas as pd
import numpy as np

# Load predictions
df = pd.read_csv('company_frequency_predictions.csv')

# Option 1: Look up a known company
company_name = 'DigitalSoft Ventures Co.'
result = df[df['company_name'] == company_name]
print(f"Frequency: {result['predicted_frequency_final'].values[0]:.3f}")
print(f"Risk Score: {result['risk_score'].values[0]:.1f}")

# Option 2: Make prediction for new company (use formulas from coefficients)
log_revenue = np.log1p(50e9)  # $50B
log_employees = np.log1p(100e3)  # 100K
is_public = 1
size_encoded = 3  # Enterprise

# Coefficients from model_coefficients.csv
beta_intercept = -0.3326
beta_revenue = 0.0478
beta_employees = -0.0512
beta_public = -0.0094
beta_size = -0.0405

# Calculation
log_lambda = (beta_intercept + 
              beta_revenue * log_revenue + 
              beta_employees * log_employees + 
              beta_public * is_public + 
              beta_size * size_encoded)

predicted_frequency = np.exp(log_lambda)  # ~1.15
risk_score = (predicted_frequency / 1.1354) * 100  # ~101

print(f"Predicted Frequency: {predicted_frequency:.3f}")
print(f"Risk Score: {risk_score:.1f}")
